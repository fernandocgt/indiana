  // Função para salvar o arremate no banco de dados
  async function registrarVenda() {
    // Captura os valores dos inputs do seu formulário
    const comissao = document.querySelector('input[value="8"]').value;
    const lances = document.querySelectorAll('input[type="number"]')[1].value;
    const valorArremate = document.querySelectorAll('input[type="number"]')[2].value;
    const comprador = document.querySelector('input[placeholder="Buscar cliente..."]').value;

    try {
      // Envia os dados para a coleção "vendas" no Firestore
      const docRef = await addDoc(collection(db, "vendas"), {
        lote: "042", // Exemplo fixo baseado no seu HTML atual
        comissao: comissao,
        quantidadeLances: lances,
        valorTotal: valorArremate,
        comprador: comprador,
        dataVenda: serverTimestamp()
      });

      alert("Venda registrada com sucesso no sistema Indiana!");
      console.log("ID da Transação: ", docRef.id);
    } catch (error) {
      console.error("Erro ao salvar no Firestore: ", error);
      alert("Erro ao conectar com o banco de dados.");
    }
  }

  document.querySelector('button.bg-primary').addEventListener('click', registrarVenda);
